  const loadAllDashboardData = async (apiKey: string) => {
    if (!apiKey) return
    setRefreshing(true)
    
    try {
      // Fetch dashboard stats
      const statsResponse = await fetch(`/api/customer/dashboard?apiKey=${apiKey}&type=stats`)
      const statsResult = await statsResponse.json()
      
      if (statsResult.success) {
        setDashboardData({
          stats: statsResult.stats,
          loading: false,
          onRefresh: () => loadAllDashboardData(apiKey)
        })
      } else {
        console.error('Failed to load dashboard stats:', statsResult.error)
        message.error('Failed to load dashboard data')
      }
    } catch (error) {
      console.error('Dashboard data fetch error:', error)
      message.error('Failed to fetch dashboard data')
    } finally {
      setRefreshing(false)
    }
  }
